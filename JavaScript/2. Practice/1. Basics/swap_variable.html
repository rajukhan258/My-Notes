JavaScript Program to Swap Two Variables

Example 1: Using a Temporary Variable


        

            
<script>
    let a = prompt(`Enter a number`);
    document.getElementById("var1").innerHTML = `Let a = ${a}`;
    let b = prompt(`Enter a name/varibale`);
    document.getElementById("var2").innerHTML = `Let b = ${b}`;;

    let temp = a;
    a = b;
    b = temp;

    document.getElementById("swap").innerHTML = `The value of a after swapping: 
    a = ${a}  b = ${b}`;
</script>
            

        

    





    

        
Example 2: Using es6(ES2015) Destructuring assignment


        

        

            
<script>
    let a = prompt(`Enter a number`);

    let b = prompt(`Enter a name/varibale`);

Example 3: Using Arithmetic Operators


        

        

            
<script>
    let a = prompt(`Enter a number`);
    document.getElementById("var1").innerHTML = `Let a = ${a}`;
    let b = prompt(`Enter a name/varibale`);
    document.getElementById("var2").innerHTML = `Let b = ${b}`;;

a = a + b;
b = a - b;
a = a - b;
    document.getElementById("swap").innerHTML = `The value of a after swapping: 
    a = ${a}  b = ${b}`;
</script>
            

        

    




    

        
Example 3: Using Bitwise XOR operator


        

        

            
<script>
    let a = prompt(`Enter a number`);
    document.getElementById("var1").innerHTML = `Let a = ${a}`;
