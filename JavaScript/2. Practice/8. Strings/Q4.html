Q. Write a function that counts the occurrences of a specific character in a string.

Example

<script>
    // Function to count occurrences of a specific character in a string
    function countOccurrences(str, char) {
        let count = 0;
        for (let i = 0; i < str.length; i++) {
            if (str[i] === char) {
                count++;
            }
        }
        return count;
    }

    // Input sentence and character to count
    let sentence = "The energetic engineer eagerly explored the eerie cave."
    let chatToCount = "e";

    // Counting occurrences of the specified character in the sentence
    let occurrences = countOccurrences(sentence, chatToCount);

    // Displaying the count of occurrences in an HTML element
    document.getElementById("output-1").innerHTML = `The character '${chatToCount}' occurs ${occurrences} times.`;
</script>
