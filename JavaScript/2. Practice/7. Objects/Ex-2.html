Object Exercise

Exercises: Level 2

  1. Find the person who has many skills in the users object.
  2. Count logged in users, count users having greater than equal to 50 points from the following object.
  3. Find people who are MERN stack developer from the users object
  4. Set your name in the users object without modifying the original users object
  5. Get all keys or properties of users object

Example

<script>
    // The users object containing user information
    const users = {
        Alex: {
            email: 'alex@alex.com',
            skills: ['HTML', 'CSS', 'JavaScript'],
            age: 20,
            isLoggedIn: false,
            points: 30
        },
        Asab: {
            email: 'asab@asab.com',
            skills: ['HTML', 'CSS', 'JavaScript', 'Redux', 'MongoDB', 'Express', 'React', 'Node'],
            age: 25,
            isLoggedIn: false,
            points: 50
        },
        Brook: {
            email: 'daniel@daniel.com',
            skills: ['HTML', 'CSS', 'JavaScript', 'React', 'Redux'],
            age: 30,
            isLoggedIn: true,
            points: 50
        },
        Daniel: {
            email: 'daniel@alex.com',
            skills: ['HTML', 'CSS', 'JavaScript', 'Python'],
            age: 20,
            isLoggedIn: false,
            points: 40
        },
        John: {
            email: 'john@john.com',
            skills: ['HTML', 'CSS', 'JavaScript', 'React', 'Redux', 'Node.js'],
            age: 20,
            isLoggedIn: true,
            points: 50
        },
        Thomas: {
            email: 'thomas@thomas.com',
            skills: ['HTML', 'CSS', 'JavaScript', 'React'],
            age: 20,
            isLoggedIn: false,
            points: 40
        },
        Paul: {
            email: 'paul@paul.com',
            skills: ['HTML', 'CSS', 'JavaScript', 'MongoDB', 'Express', 'React', 'Node'],
            age: 20,
            isLoggedIn: false,
            points: 40
        }
    };

    // Initializing variables to store specific user information
    let maxSkills = 0; // Variable to track maximum skills
    let skillfulPerson = ''; // Variable to store the person with the most skills
    let loggedInusers = 0; // Variable to count logged-in users
    let highPointsUsers = 0; // Variable to count users with points >= 50
    const mernDevelopers = []; // Array to store MERN stack developers

    // Looping through the users object to find specific information
    for (let keyuser in users) {
        // Finding the person with the most skills
        if (users[keyuser].skills.length > maxSkills) {
            maxSkills = users[keyuser].skills.length;
            skillfulPerson = keyuser;
        }
        // Counting logged-in users and users with points >= 50
        if (users[keyuser].isLoggedIn && users[keyuser].points >= 50) {
            loggedInusers++;
            highPointsUsers++;
        }

        // Identifying MERN stack developers
        const skills = users[keyuser].skills;
        if (skills.includes("MongoDB") &&
            skills.includes("Express") &&
            skills.includes("React") &&
            skills.includes("Node")) {
            mernDevelopers.push(keyuser);
        }
    }

    // Displaying information on the HTML page
    document.getElementById("output-1").innerHTML = `<b>Ans 1:</b> Most skillful Person: ${skillfulPerson}`;

    document.getElementById("output-2").innerHTML = `<b>Ans 2:</b> Logged-in users: ${loggedInusers}<br>Users with >= 50 points: ${highPointsUsers}`;

    document.getElementById("output-3").innerHTML = `<b>Ans 3:</b> MERN stack developers: ${mernDevelopers.join(", ")}`;

    // Adding a new user 'Joseph' to the users object
    const YourName = {
        ...users, Joseph: {
            email: 'Joseph@John.com',
            skills: ['HTML', 'CSS', 'JavaScript'], // 
            age: 25,
            isLoggedIn: true,
            points: 40
        }
    };

    // Generating the object structure with its properties and values
    let output = document.getElementById("output-4");
    let text = "{<br>";
    for (let key in YourName) {
        text += `${key}: {`;
        for (let prop in YourName[key]) {
            // Checking and formatting string values
            if (prop === 'value' && typeof YourName[key][prop] === 'string') {
                text += `${prop}: "${YourName[key][prop]}", `;
            } else {
                text += `${prop}: ${YourName[key][prop]}, `;
            }
        }
        text = text.slice(0, -2);
        text += "},<br>";
    }
    text = text.slice(0, -5);
    text += "<br>}";
    output.innerHTML = `<b>Ans 4:</b> ${text}`;

    // Getting all keys/properties of the users object and displaying them
    const Userkeys = Object.keys(users);
    document.getElementById("output-5").innerHTML = `<b>Ans 5:</b> [${Userkeys.join(", ")}]`;
</script>
