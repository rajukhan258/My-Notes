Object Exercise:

Exercises: Level 3

Create an object literal called personAccount. It has firstName, lastName, incomes, expenses properties and it has totalIncome, totalExpense, accountInfo,addIncome, addExpense and accountBalance methods.

Incomes is a set of incomes and its description and expenses is a set of incomes and its description.

Example

<script>
    // Creating a personAccount object
    const personAccount = {
        firstName: "John",
        lastName: "Smith",
        incomes: [],
        expenses: [],

        // Method to calculate total income
        totalIncome: function () {
            return this.incomes.reduce(function (totalincome, income) {
                return totalincome + income.amount;
            }, 0);
        },

        // Method to add income
        addIncome: function (description, amount) {
            this.incomes.push({ description, amount });
        },

        // Method to calculate total expense
        totalExpense: function () {
            return this.expenses.reduce(function (totalexpense, expense) {
                return totalexpense + expense.amount;
            }, 0);
        },

        // Method to add expense
        addExpense: function (description, amount) {
            this.expenses.push({ description, amount });
        },

        // Method to calculate account balance
        accountBalance: function () {
            return this.totalIncome() - this.totalExpense();
        },

        // Method to display account information
        accountInfo: function () {
            return `<b>Account Information:</b><br>
Name: ${this.firstName} ${this.lastName}<br>
Total Income: ₹${this.totalIncome()}<br>
Total Expense: ₹${this.totalExpense()}<br>
Balance: ₹${this.accountBalance()}`;
        },
    };

    // Adding income and expense details
    personAccount.addIncome("Salary", 3000);
    personAccount.addIncome("Freelancing", 500);
    personAccount.addExpense("Rent", 1000);
    personAccount.addExpense("Groceries", 300);

    // Displaying account information on the HTML element with id "output-1"
    document.getElementById("output-1").innerHTML = `${personAccount.accountInfo()}`;
</script>
