Object Methods

Exercises: Level 4

1. Imagine you are getting users collection from a MongoDB database.

  1. Create a function called signUp which allows user to add to the collection. If user exists, inform the user that he has already an account.
  2. Create a function called signIn which allows user to sign in to the application

2. The products array has three elements and each of them has six properties.

  1. Create a function called rateProduct which rates the product
  2. Create a function called averageRating which calculate the average rating of a product

3. Create a function called likeProduct. This function will helps to like to the product if it is not liked and remove like if it was liked.

    // Function to register a user
    function signUp(username, email, password) {
        // Checking if the user already exists based on their email
        const userExists = users.find(function (users) {
            return users.email === email;
        });
        if (userExists) {
            return "User already registered!";
        } else {
            // Creating a new user object and adding it to the users array
            const newUser = {
                _id: Math.random().toString(36).substr(2, 9),
                usename: username,
                email: email,
                password: password,
                createdAt: new Date().toLocaleString(),
                isLoggedIn: false
            };
            users.push(newUser);
            return "User sucessfully created!";
        }
    }

    // Function to log in a user
    function signIn(email, password) {
        // Finding the user by email and password match
        const user = users.find(function (user) {
            return user.email === email && user.password === password;
        })
        if (user) {
            return "User successfully loggedIn";
        } else {
            return "Invalid credentials";
        }
    }

    // Function to rate a product
    function rateProduct(productId, userId, rating) {
        // Finding the product by its ID
        const product = products.find(function (Arg_product) {
            return Arg_product._id === productId;
        });
        if (product) {
            // Adding the user's rating to the product
            product.ratings.push({ userId, rate: rating });
            return "Product rated succesfully!";
        } else {
            return "Product not found!";
        }
    }
    // Function to calculate the average rating of a product
    function averageRating(productId) {
        const product = products.find(function (product) {
            return product._id === productId;
        });
        if (product) {
            // Calculating the average rating
            const totalRatings = product.ratings.reduce(function (sum, rating) {
                return sum + rating.rate;
            }, 0);
            const average = totalRatings / product.ratings.length || 0;
            return `Average rating for '${product.name}' is ${average.toFixed(2)}`;
        } else {
            return "Product not found!";
        }
    }

    // Displaying the outputs of rateProduct and averageRating functions
    document.getElementById("output-2").innerHTML = `${rateProduct('aegfal', 'thomas123', 4)}<br>
                ${averageRating('aegfal')}`;


    // Function to like/unlike a product
    function likeProduct(productId, userId) {
        // Finding the product by its ID
        const product = products.find(function (prod) {
            return prod._id === productId;
        });

        if (product) {
            // Check if the user has already liked the product
            const likedIndex = product.likes.indexOf(userId);

            if (likedIndex === -1) {
                // If the user hasn't liked the product, add the userId to likes array
                product.likes.push(userId);
                return 'Product liked successfully!';
            } else {
                // If the user has already liked the product, remove the like
                product.likes.splice(likedIndex, 1);
                return 'Product unliked successfully!';
            }
        } else {
            // If the product is not found
            return 'Product not found!';
        }
    }

    // Displaying the outputs of likeProduct function
    document.getElementById("output-3").innerHTML = `${likeProduct('aegfal', 'ghderc')}<br> ${likeProduct('aegfal', 'ghderc')}`;
