Check senior citizen or not && eligible for vote or not

Q5. Input Name and Age of a person
        a)Check he is senior citizen or not
        b)Check he is eligible for vote or not

            
<script>
    const na = prompt("Enter a Name");
    document.getElementById("name").innerHTML = (`Name: ${na}`);
    const age = parseInt(prompt("Enter age"));
    document.getElementById("age").innerHTML = (`Age: ${age}`);

    if (age >= 60) {
        document.getElementById("result").innerHTML =(`${na} is a Senior citizen and eligible for vote.`);
    }

    else if (age >= 18 && age <= 59) {
        document.getElementById("result").innerHTML =(`${na} is not a Senior citizen but eligible for vote.`);;
    }

    else {
        document.getElementById("result").innerHTML =(`${na} is not a Senior citizen, also he is not eligible for vote.`);;
    }
</script>
