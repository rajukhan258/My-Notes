Program to Check Prime Number

CODE USED:

        
<script>
// take input from the user
const number = parseInt(prompt("Enter a positive number: "));
document.getElementById("input").innerHTML = `Given number = ${number}`;
let isPrime = true;

// check if number is equal to 1
if (number === 1) {
    document.getElementById("result").innerHTML = ("1 is neither prime nor composite number.");
}
// check if number is greater than 1
else if (number > 1) {

    // looping through 2 to number-1
    for (let i = 2; i < number; i++) {
        if (number % i == 0) {
            isPrime = false;
            break;
        }
    }
    if (isPrime) {
        document.getElementById("result").innerHTML =(`${number} is a prime number.`);
    } else {
        document.getElementById("result").innerHTML =(`${number} is a not prime number.`);
    }
}
// check if number is less than 1
else {
    document.getElementById("result").innerHTML =("Please enter a positive number.");
}  
</script>
