Program to Find the Factorial of a Number

        
<script>
const num = parseInt(prompt('Enter a positive integer: '));

if (num < 0) {
    document.write(`Factorial is not defined for negative numbers`);
}
else if (num === 0) {
    document.write(`${num}! = ${1}`); // factorial of zero equals one!
}
else {
    let fact = 1;
    for (i = 1; i <= num; i++) {
        fact = fact * i;    //fact *= i;
    };

    document.getElementById("result").innerHTML = (`${num}! =  ${fact}`);
}
</script>
