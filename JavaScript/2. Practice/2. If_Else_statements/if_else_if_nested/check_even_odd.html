Program to check Even/ Odd

Code used

    
<script>
    let num = parseInt(prompt("Enter an Interger"));
    document.getElementById("int").innerHTML = `Given Number: ${num}`;

    if (num >= 1 && num % 2 == 0) {
        document.getElementById("result").innerHTML = `${num} is a positive and even number.`;
    }
    else if (num >= 1 && num % 2 != 0) {
        document.getElementById("result").innerHTML = `${num} is a positive and odd number.`;
    }
    else if (num == 0) {
        document.getElementById("result").innerHTML = `${num} is neither positive nor negative.`;
    }
    else {
        document.getElementById("result").innerHTML = `Please enter a positive number to check Even And Odd.`;
    }
</script>
