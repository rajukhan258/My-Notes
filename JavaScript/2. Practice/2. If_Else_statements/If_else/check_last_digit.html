Check if the Numbers Have Same Last Digit



<script>
    const first_num = parseInt(prompt(`Enter Firtst Number:`));
    const second_num = parseInt(prompt(`Enter Second Number:`));
    const third_num = parseInt(prompt(`Enter Third Number:`));

    const remainder_1 = first_num % 10;
    const remainder_2 = second_num % 10;
    const remainder_3 = third_num % 10;

    if(remainder_1 == remainder_2 && remainder_1 == remainder_3){
        document.write(`${first_num}, ${second_num} and ${third_num} have the same last digit.`);
    }
    else{
        document.write(`${first_num}, ${second_num} and ${third_num} have different last digit.`);
    }
</script>
