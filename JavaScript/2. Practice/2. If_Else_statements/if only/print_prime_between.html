Print All Prime Numbers in an Interval

            
<script>
// take input from the user
const startnum = parseInt(prompt("Enter Start number"));
const endnum = parseInt(prompt("Enter End number"));

// looping from first number to last number
for (i = startnum; i <= endnum; i++) {
    let isPrime = true;  //let flag = 0;

    //looping through 2 to user input number
    for (let j = 2; j < i; j++) {
        if (i % j == 0) {
            isPrime = false;    //flag = 1;
            break;
        }
    }

    //if number greater than 1 and not divisible by other numbers
    if (i > 1 && isPrime == true) {
        document.write(`&emsp;` + i + ' ');
    }
}
</script>
