Returning different values based on a conditons:

<script>
    let score = prompt("Enter score:");
    let grade = (score > 100)? "Invalid Number" : (score >= 90) ? "A+" : (score >= 80) ? "A" : (score >=70) ? "B" : (score >= 60) ? "C" : "F";
    document.getElementById('output').innerHTML = `Your grade is: <b>${grade}</b>`;
</script>
