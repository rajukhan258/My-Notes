Choosing between two different actions based on a condition:

<script>
    let x=  prompt("Enter first number");
    let y=  prompt("Enter second number");
    let action = (x > y) ? function(){document.getElementById("output").innerHTML = "<b>x</b> id greater than <b>y</b>."} : function(){document.getElementById("output").innerHTML = "<b>y</b> id greater than <b>x</b>."};

    action();
</script>
