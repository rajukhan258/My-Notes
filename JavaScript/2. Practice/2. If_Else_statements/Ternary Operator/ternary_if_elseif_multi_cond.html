Ternary operator with multiple conditions

<script>
    let year = parseInt(prompt("Enter a Year to check Leap year:"));
    let result = (year % 4 == 0 && year % 100 !== 0) ? "leap Year" : (year % 400 == 0) ? "leap Year" : "Not a leap Year";
    document.getElementById("output").innerHTML = `Year <b>${year}</b> is a <b>${result}</b>. `
</script>
