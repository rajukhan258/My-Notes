Perform arithmetic operations on two numbers



<script>
    function operation(num1, num2, operator){
        if(OP == "+"){
            document.write(`Sum of ${num1} and ${num2} is ${num1+num2}`);
        }
        else if(OP == "-"){
            document.write(`Difference of ${num1} and ${num2} is ${num1-num2}`);
        }
        else if(OP == "*"){
            document.write(`Product of ${num1} and ${num2} is ${num1*num2}`);
        }
        else if(OP == "/"){
            document.write(`Division of ${num1} and ${num2} is ${num1/num2}`);
        }
        else if(OP == "%"){
            document.write(`Modulus of ${num1} and ${num2} is ${num1%num2}`);
        }
        else{
            document.write("ERROR Occured")
        }
    }

    let x = parseFloat(prompt(`Enter the first number`));
    let y = parseFloat(prompt(`Enter the second number`));
    let OP = prompt(`Enter a Operator: "+", "-", "*", "/", "%"`);

    operation(x, y, OP)
 </script>
