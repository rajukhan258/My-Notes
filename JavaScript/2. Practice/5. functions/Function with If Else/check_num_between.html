Find the number is present in given range



<script>
    function checkInRange(num, start, end){
        if(num >= start && num <= end){
            document.write(`${num} is between the range ${start} and ${end}.`)
        }
        else{
            document.write(`${num} is outside the range ${start} and ${end}.`)
        }
    }

    let num = parseInt(prompt("Enter the number you want to check."));
    let start = parseInt(prompt("Enter the starting lower number of range."));
    let end = parseInt(prompt("Enter the last upper number of the range."));

    checkInRange(num, start, end);
</script>
