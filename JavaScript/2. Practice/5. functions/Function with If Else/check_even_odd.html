Check if a number is odd or even



<script>
    function isEvenorOdd(num){
        if(num % 2 == 0){
            document.write(`${num} is a Even number.`);
        }
        else{
            document.write(` ${num} is an Odd Number.`);
        }
    }
    let myNumber = prompt("Enter your number");
    isEvenorOdd(myNumber);
</script>
