Find the largest of three number



<script>
    function findLargest(num1, num2, num3) {
        if (num1 >= num2 && num1 >= num3) {
            document.write(`${num1} is the largest number.`);
        } else if (num2 >= num3 && num2 >= num1) {
            document.write(` ${num2}  is the largest number.`);
        } else {
            document.write(`${num3} is the largest number.`);
        }
    }
    var a = parseFloat(prompt("Enter the first number"));
    var b = parseFloat(prompt("Enter the second number"));
    var c = parseFloat(prompt("Enter the third number"));

    findLargest(a, b, c);
</script>
