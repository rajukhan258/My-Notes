Check if input variable is a number or not

isNaN() method returns true if a value is Not-a-Number.

Number.isNaN() returns true if a number is Not-a-Number.

In other words:
isNaN() converts the value to a number before testing it.



<script>
    function isValidNumber(num){
        if(isNaN(num)){
            document.write(`isNaN (Not-a-Number): ${isNaN(num)}<br> `);
            document.write(`<b>${num}</b> is not a number.`);
        }
        else{
            document.write(`isNaN (Not-a-Number): ${isNaN(num)}<br> `);
            document.write(`<b>${num}</b> is a valid number.`);
        }
    }
    let mynum = prompt("Enter a Number");
    isValidNumber(mynum);
</script>
