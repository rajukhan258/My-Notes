Find the grade for input marks

  • Print “S grade” if marks is between 90 and 100.
  • Print “A grade” if marks is between 80 and 90.
  • Print “B grade” if marks is between 70 and 80.
  • Print “C grade” if marks is between 60 and 70.
  • Print “D grade” if marks is between 50 and 60.
  • Print “E grade” if marks is between 40 and 50.
  • Print “Student has failed” if marks is between 0 and 40.
  • Else print “Marks can't be more than 100;”. 
<script>
    function printGrade(name, marks){
        if (marks >= 90 && marks <= 100){
            document.write(`${name}, Grade: S.`);
        }
        else if (marks >= 80 && marks < 90){
            document.write(`${name}, Grade: A.`); 
        }
        else if (marks >= 70 && marks < 80){
            document.write(`${name}, Grade: B.`); 
        }
        else if (marks >= 60 && marks < 70){
            document.write(`${name}, Grade: C.`); 
        }
        else if (marks >= 50 && marks <60){
            document.write(`${name}, Grade: D.`); 
        }
        else if (marks >= 40 && marks <50){
            document.write(`${name}, Grade: E.`); 
        }
        else if (marks >= 0 && marks <40){
            document.write(`${name} has Failed.`); 
        }
        else{
            document.write(`Marks can't be more than 100;`); 
        }
    }

    let name = prompt("Enter Student name.");
    let marks = parseFloat(prompt("Enter marks obtained."))

    printGrade(name, marks);
</script>
