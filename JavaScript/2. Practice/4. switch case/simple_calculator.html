Simple Calculator



<script>
    const operator = prompt("Enter operator: (either +, -, *, /)");
    const num1 = parseFloat(prompt(("Enter first number")));
    const num2 = parseFloat(prompt(("Enter second number")));

    switch (operator){
        case "+":
            document.write(`The sum of ${num1} and ${num2} is : 
            ${num1+num2}`);
            break;
        case "-":
            document.write(`The difference between
            ${num1} and ${num2} is :  ${num1-num2}`) ;
            break;
        case "*":
            document.write(`The product of
            ${num1} and ${num2} is :   ${num1*num2}`
            );
            break;
        case "/":
            document.write(`The product of
            ${num1} and ${num2} is :   ${num1/num2}`
            );
            break;
        default:{
            alert('Invalid Operator');
            };
    }
</script>
