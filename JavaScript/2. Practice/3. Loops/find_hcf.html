Program to Find HCF or GCD



<script>
    const num_1 = parseInt(prompt(`Enter first number: `));
    const num_2 = parseInt(prompt(`Enter second number: `));

    let hcf;
    for(i = 1; i <= num_1 && i <= num_2; i++){
        if(num_1 % i == 0 && num_2 % i == 0){
            hcf  = i;
        }         
    }
    document.write(`HCF of ${num_1} and ${num_2} is ${hcf}.`)
</script>
