Program to Find LCM



<script>
    const number_1 = parseInt(prompt("Enter first number: "));
    const number_2 = parseInt(prompt("Enter second number: "));

    let hcf;
    for(i = 1; i <= number_1 && i <= number_2; i++){
        if(number_1 % i == 0 && number_2 % i == 0){
            hcf = i;
        }
    }

    let lcm = (number_1 * number_2) / hcf;
    document.write(`LCM of ${number_1} and ${number_2} is ${lcm}.`)
</script>
