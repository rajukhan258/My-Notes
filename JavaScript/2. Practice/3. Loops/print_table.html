Program to Display the Multiplication Table

For Loop

While Loop

Do-While Loop

For loop

        
<script>
    for (count = 1; count <= 10; count++) {
        document.write(num + " * " + count + " = " + num * count + "<br>");
    }
</script>

While loop

        
<script>
    count = 1;
    while (count <= 10) {
        document.write(num + " * " + count + " = " + num * count + "
");
        count++;
    }
</script>

Do-While loop

        
<script>
    let dowhilecount = 1;
    do {
        document.write(num + " * " + dowhilecount + " = " + num * dowhilecount + "
");
        dowhilecount++;
    }
</script>
