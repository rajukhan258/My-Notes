Arrays Exercise

Question

First remove all the punctuations and change the string to array and count the number of words in the array

Example

<script>
    let text = `I love teaching and empowering people. I teach HTML, CSS, JS, React,`;
    const wordsArray = text.replace(/[.,]/g, '').split(/\s+/);

    function run() {
        document.getElementById("p1").innerHTML = arrprint(wordsArray);
        function arrprint(arr) {
            let output = "[";
            for (i = 0; i < arr.length; i++) {
                if (Array.isArray(arr[i])) {
                    output += arrprint(arr[i]);
                }
                else if (typeof arr[i] === 'string') {
                    output += `"${arr[i]}"`;
                }
                else {
                    output += `${arr[i]}`;
                }
                if (i < arr.length - 1) {
                    output += ", ";
                }
            }
            output += "]";
            return output;
        }
        document.getElementById("p2").innerHTML = `Length of Array: ${wordsArray.length}`;
    };
</script>
