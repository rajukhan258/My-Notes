Questions

In the following shopping cart add, remove, edit items

const shoppingCart = ['Milk', 'Coffee', 'Tea', 'Honey']
  • add 'Meat' in the beginning of your shopping cart if it has not been already added
  • add Sugar at the end of you shopping cart if it has not been already added
  • remove 'Honey' if you are allergic to honey
  • modify Tea to 'Green Tea'

Example

<script>
    const shoppingCart = ['Milk', 'Coffee', 'Tea', 'Honey'];

    //add 'Meat' in the beginning of your shopping cart if it has not been already added
    const Ans_1 = shoppingCart.includes("Meat");
    if (Ans_1 === true) {
        document.getElementById("output-1").innerHTML = `Meat already exits in shopping Cart.`;
    }
    else {
        shoppingCart.unshift("Meat");
        document.getElementById("output-1").innerHTML = `[${shoppingCart.join(", ")}]`;
    }

    //add Sugar at the end of you shopping cart if it has not been already added
    const Ans_2 = shoppingCart.includes("Sugar")
    if (Ans_2 === true) {
        document.getElementById("output-2").innerHTML = `Sugar already exits in shopping Cart.`;
    }
    else {
        shoppingCart.push("Sugar");
        document.getElementById("output-2").innerHTML = `[${shoppingCart.join(", ")}]`;
    }

    //remove 'Honey' if you are allergic to honey
    const Ans_3 = shoppingCart.splice(4, 1);
        document.getElementById("output-3").innerHTML = `[${shoppingCart.join(", ")}]`;

</script>
