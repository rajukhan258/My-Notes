Array Exercise

Question 2:

  1. Declare an array variable name itCompanies and assign initial values Facebook, Google, Microsoft, Apple, IBM, Oracle and Amazon
  2. Print the array using document.getElementbyID()
  3. Print the number of companies in the array
  4. Print the first company, middle and last company
  5. Print out each company
  6. Change each company name to uppercase one by one and print them out
  7. Print the array like as a sentence: Facebook, Google, Microsoft, Apple, IBM,Oracle and Amazon are big IT companies.
  8. Check if a certain company exists in the itCompanies array. If it exist return the company else return a company is not found
  9. Sort the array using sort() method
  10. Reverse the array using reverse() method
  11. Slice out the first 3 companies from the array
  12. Slice out the last 3 companies from the array
  13. Slice out the middle IT company or companies from the array
  14. Remove the first IT company from the array
  15. Remove the middle IT company or companies from the array
  16. Remove the last IT company from the array
  17. Remove all IT companies

Solutions:

<script>
    const itCompanies = ["Facebook", "Google", "Microsoft", "Apple", "IBM", "Oracle", "Amazon",];

    // Print the array
    document.getElementById("ans2").innerHTML = `<b>Ans 2:</b> [${itCompanies.join(", ")}]`;

    // Print the number of companies
    document.getElementById("ans3").innerHTML = `<b>Ans 3:</b> No. of companies = ${itCompanies.length}`;

    // Print the first, middle, and last companies

    document.getElementById("ans4").innerHTML = `<b>Ans 4:</b> First Company = "${itCompanies[0]}" <br> 
    Middle Company = "${itCompanies[Math.floor(itCompanies.length / 2)]}" <br> 
    Last Company =  "${itCompanies[itCompanies.length - 1]}"`;

    // Print out each company
    document.getElementById("ans5").innerHTML = `<b>Ans 5: </b> ${itCompanies.join("<br> ")}`;

    // Change each company name to uppercase 
    let output6 = document.getElementById("ans6");
    output6.innerHTML = `<b>Ans 6:</b>`;
    itCompanies.forEach(elements => {
        { output6.innerHTML += elements.toUpperCase() + "<br>" }
    });

    // Print the array as a sentence
    let output7 = document.getElementById("ans7");
    output7.innerHTML = `<b>Ans 7: </b>${itCompanies.join(", ")} are big iT companies.`;

    // Check if a certain company exists in the itCompanies array
    let output8 = document.getElementById("ans8");
    output8.innerHTML = `<b>Ans 8: </b>`;
    if (itCompanies.includes("Facebook")) {
        output8.innerHTML += "Facebook found";
    }
    else {
        output8.innerHTML += "Not Found"
    };

    // Sort the array
    itCompanies.sort();
    document.getElementById("ans9").innerHTML = `<b>Ans 9: </b>[${itCompanies.join(", ")}]`;

    // Reverse the array
    itCompanies.reverse();
    document.getElementById("ans10").innerHTML = `<b>Ans 10: </b>[${itCompanies.join(", ")}]`;

    // Slice out the first 3 companies
    const first3Companies = itCompanies.slice(0, 3);
    document.getElementById("ans11").innerHTML = `<b>Ans 11: </b>[${first3Companies.join(", ")}]`;

    // Slice out the last 3 companies\
    const last3Companies = itCompanies.slice(-3);
    document.getElementById("ans12").innerHTML = `<b>Ans 12: </b>[${last3Companies.join(", ")}]`;

    // Slice out the middle IT company or companies
    const middleIndex = Math.floor(itCompanies.length / 2);
    const isEven = itCompanies.length % 2 === 0;
    let middleCompanies;
    if (isEven) {
        middleCompanies = [itCompanies[middleIndex - 1], itCompanies[middleIndex]];
    }
    else {
        middleCompanies = [itCompanies[middleIndex]];
    }
    document.getElementById("ans13").innerHTML = `<b>Ans 13: </b>[${middleCompanies.join(", ")}]`;

    // Remove the first IT company
    itCompanies.shift();
    document.getElementById("ans14").innerHTML = `<b>Ans 14: </b>[${itCompanies.join(", ")}]`;

    // Remove the middle IT company or companies
    itCompanies.splice(middleIndex - 1, middleCompanies.length + 1);
    document.getElementById("ans15").innerHTML = `<b>Ans 15: </b>[${itCompanies.join(", ")}]`;

    // Remove the last IT company
    itCompanies.pop()
    document.getElementById("ans16").innerHTML = `<b>Ans 16: </b>[${itCompanies.join(", ")}]`;

    // Remove all IT companies
    itCompanies.splice(0, itCompanies.length)
    document.getElementById("ans17").innerHTML = `<b>Ans 17: </b>[${itCompanies.join(", ")}]`;
</script>
