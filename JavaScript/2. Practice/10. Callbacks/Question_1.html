Callback practice Questions

Quesetion: Write a function that uses a synchronous callback to double each number in an array.

Example

<script>
	// Function to double each number in an array using a callback
	function doubleNumbers(arr, callback) {
		const doubledArray = [];
		for (let i = 0; i < arr.length; i++) {
			// Apply the callback function to each element and store the doubled value
			const doubledNumbers = callback(arr[i]);
			doubledArray.push(doubledNumbers); // Push the doubled value to the new array
		}
		return doubledArray; // Return the array with doubled values
	}

	// Callback function to double a number
	function double(num) {
		return num * 2;
	}

	// Event listener for a button click (ID: 'runcode')
	document.getElementById('runcode').addEventListener('click', function () {
		// Array of numbers
		const numbers = [1, 2, 3, 4, 5];

		// Double each number using the doubleNumbers function with the 'double' callback
		const doubled = doubleNumbers(numbers, double);

		document.getElementById('output-1').innerHTML = doubled.join(", ");
	});
</script>
