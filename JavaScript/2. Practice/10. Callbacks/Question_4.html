Callback practice Questions

Question_4: Develop a function that uses setInterval and an asynchronous callback to print numbers from 1 to 5 at intervals of 1 second.

Example

<script>
	// Function to print numbers asynchronously at intervals
	function toPrint(callback) {
		let count = 1;
		const interval_id = setInterval(() => {
			if (count > 5) {
				clearInterval(interval_id); // Clear the interval when count exceeds 5
			} else {
				callback(count); // Invoke the provided callback with the current count
				count++;
			}
		}, 1000); // Interval set to 1 second (1000 milliseconds)
	}

	// Callback function that appends the number to an HTML element
	function forcallback(num) {
		document.getElementById('output-1').innerHTML += `${num}<br>`;
	}

	// Adding an event listener to execute the 'toPrint' function with 'forcallback' as a callback
	document.getElementById('runcode').addEventListener('click', function () {
		// Invoking the 'toPrint' function with the 'forcallback' as a callback
		const result = toPrint(forcallback);
	});
</script>
