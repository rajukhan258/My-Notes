Callback practice Questions

Question_3: Define a function that takes an array of strings and a synchronous callback to concatenate them into a single sentence.

Example

<script>
	// Function to concatenate an array of strings into a single sentence using a callback
	function concatenateStrings(array, callback) {
		let concateningSentence = ''; // Initialize an empty string to store the concatenated sentence
		for (let i = 0; i < array.length; i++) {
			concateningSentence = callback(concateningSentence, array[i]); // Apply the callback to concatenate strings
		}
		return concateningSentence; // Return the concatenated sentence
	}

	// Callback function to concatenate a sentence and a word with a space
	function concatenate(sentence, word) {
		return `${sentence} ${word}`; 
	}

	// Event listener for a button click (ID: 'runcode')
	document.getElementById('runcode').addEventListener('click', function () {
		const arrOfString = ['This', 'is', 'a', 'sample', 'sentence.']; // Array of strings

		// Concatenate the array of strings into a single sentence using the concatenateStrings function with the 'concatenate' callback
		const concatenatedSentence = concatenateStrings(arrOfString, concatenate);

		// Display the concatenated sentence in the HTML element with ID 'output-1'
		document.getElementById('output-1').innerHTML = concatenatedSentence;
	});
</script>
